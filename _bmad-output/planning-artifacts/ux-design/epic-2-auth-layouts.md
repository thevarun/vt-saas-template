/* Pseudo-elements for blur effect */
.bg-decoration::before {
  content: '';
  position: absolute;
  top: -10%;
  left: -10%;

  height: 40%;
  background: rgb(219 234 254 / 0.5); /* blue-100/50 */

  content: '';
  position: absolute;
  bottom: -10%;
  right: -10%;
